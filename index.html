// Weather data fetching with error handling
async function fetchWeatherData(city) {
  try {
    const response = await fetch(
      `${API_BASE_URL}?q=${city}&appid=${API_KEY}&units=metric`
    );
    if (!response.ok) throw new Error('City not found');
    
    const data = await response.json();
    return {
      temperature: data.main.temp,
      humidity: data.main.humidity,
      description: data.weather[0].description,
      icon: data.weather[0].icon
    };
  } catch (error) {
    console.error('Weather fetch failed:', error);
    return null;
  }
}

# Carbon footprint calculation algorithm
def calculate_transport_emissions(distance, vehicle_type):
    emission_factors = {
        'car': 0.2,      # kg CO2 per km
        'bus': 0.05,     # kg CO2 per km
        'train': 0.04,   # kg CO2 per km
        'bicycle': 0.0   # kg CO2 per km
    }
    
    factor = emission_factors.get(vehicle_type, 0.2)
    return distance * factor

// Real-time messaging component
const ChatRoom = ({ roomId, userId }) => {
  const [messages, setMessages] = useState([]);
  const [newMessage, setNewMessage] = useState('');

  useEffect(() => {
    const socket = io('/chat');
    socket.emit('join-room', roomId);
    
    socket.on('message', (message) => {
      setMessages(prev => [...prev, message]);
    });
    
    return () => socket.disconnect();
  }, [roomId]);

  const sendMessage = () => {
    socket.emit('send-message', {
      roomId,
      userId,
      text: newMessage,
      timestamp: Date.now()
    });
    setNewMessage('');
  };
};
